\begin{figure}[!htbp]
    \centering
    \begin{tikzpicture}[scale=0.35]
        \draw (0,0) circle [radius=10];
        \coordinate[label=above left:{$A$}] (A) at (-6,8);
        \coordinate[label=above left:{$C$}] (C) at (-8,6);
        \coordinate[label=above right:{$B$}] (B) at (6,8);
        \coordinate[label=above right:{$D$}] (D) at (8,6);
        \coordinate[label=above right:{$O$}] (O) at (0,0);
        \draw (A)--(B);
        \draw (C)--(D);
        \filldraw (O) circle (.1);
    \end{tikzpicture}
\end{figure}
